Boucle · for
range & listes
for i in range(5):
    print(i)  # 0..4

for f in ["pomme","kiwi"]:
    print(f)
break · continue
for i in range(1,8):
    if i==5: break
    print(i)  # 1 2 3 4
for i in range(1,8):
    if i==5: continue
    print(i)  # 1 2 3 4 6 7

Dictionnaires
Créer & modifier
d = {'nom': 'Alice', 'age': 30}
d['age'] = 31      # modifier
d['ville'] = 'Paris' # ajouter
del d['ville']      # supprimer
d.get('nom')         # → 'Alice'
Parcourir
for cle, val in d.items():
    print(cle, val)

for val in d.values():
    print(val)

Retours multiples · Lambda

    return a+b, a/b
p, d = op(10,2)  # → 12, 5.0
