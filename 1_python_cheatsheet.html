Boucle · for
range & listes
for i in range(5):  # 0 1 2 3 4
    print(i)

for f in ["pomme", "kiwi"]:
    print(f)
break · continue
for i in range(1, 8):
    if i == 5: break  # sort
    print(i)  # 1 2 3 4

for i in range(1, 8):
    if i == 5: continue # saute
    print(i)  # 1 2 3 4 6 7

Lire ligne par ligne → dict
f = open("fichier.txt", "r")
dic = {}
for line in f:
    mot = line.strip().split(',')
    dic[mot[0]] = mot[1]
print(dic)
# → {'pomme': '1.50', 'kiwi': '2.00'}
